>>> Redirection tests

> subject tests
echo -ne "mkdir subject ; cd subject ; ls -a ; ls | cat | wc -c > tutu ; cat tutu"

> Simple redirection
echo -ne "echo "Hello World" > hello.txt; cat hello.txt"

> Simple redirection 2
echo -ne "ls > list.txt; cat list.txt"

> Redirection + pipe
echo -ne "ls | wc -l > count.txt; cat count.txt"

> Redirection + pipe 2
echo -ne "touch testfile; ls | wc -l > testfile; cat testfile"

> Simple double redirection
echo -ne "echo "Hello, World!" > hello.txt ; cat < hello.txt > output.txt ; cat output.txt"

> Word count to file
echo -ne "echo "This is a test file." > testfile.txt; wc -w < testfile.txt > count.txt; cat count.txt"

> Pipe to character count
echo -ne "echo "Another test" > start.txt; cat start.txt | wc -c > chars.txt; cat chars.txt"

> Sort and output redirection
echo -ne "printf "banana\napple\norange" > fruits.txt; sort < fruits.txt > sorted_fruits.txt; cat sorted_fruits.txt"

> Sorted lines to file
echo -ne "echo "Line 1" > lines.txt; echo "Line 2" >> lines.txt; sort < lines.txt > sorted_lines.txt; cat sorted_lines.txt"

> clear file
echo -ne "rm *.txt ; rm testfile ; rm -rf subject"

>>> End
